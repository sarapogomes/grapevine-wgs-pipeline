# Entrar na pasta dos gVCFs
cd /mnt/c/Users/User/Desktop/wgs_videira/vitis_project/vcf_files_V5

# Gerar sample_map.txt com TODOS os gVCFs de uma vez
ls *.g.vcf.gz | while read f; do
    sample=$(basename "$f" .g.vcf.gz)
    echo -e "${sample}\t$(realpath $f)"
done > sample_map.txt

# Verificar
cat sample_map.txt

# Definir referência
REF="/mnt/c/Users/User/Desktop/wgs_videira/vitis_project/reference/T2T_ref/T2T_ref.fasta"

# Criar pasta temporária (evita problemas de espaço no /tmp do WSL)
mkdir -p /mnt/c/Users/User/Desktop/wgs_videira/vitis_project/tmp

gatk GenomicsDBImport \
    --sample-name-map sample_map.txt \
    --genomicsdb-workspace-path ./genomicsdb_videira \
    -L chr01 -L chr02 -L chr03 -L chr04 -L chr05 \
    -L chr06 -L chr07 -L chr08 -L chr09 -L chr10 \
    -L chr11 -L chr12 -L chr13 -L chr14 -L chr15 \
    -L chr16 -L chr17 -L chr18 -L chr19 \
    --tmp-dir /mnt/c/Users/User/Desktop/wgs_videira/vitis_project/tmp \
    --reader-threads 4 \
    2>&1 | tee genomicsdb_import.log
	
REF="/mnt/c/Users/User/Desktop/wgs_videira/vitis_project/reference/T2T_ref/T2T_ref.fasta"

gatk GenotypeGVCFs \
    -R "$REF" \
    -V gendb://./genomicsdb_videira \
    -O videira_joint.vcf.gz \
    --tmp-dir /mnt/c/Users/User/Desktop/wgs_videira/vitis_project/tmp \
    2>&1 | tee genotype_gvcfs.log

# Quantas variantes e que tipos
bcftools stats videira_joint.vcf.gz | grep "^SN"

# Ver as primeiras variantes para confirmar as 15 amostras
bcftools view videira_joint.vcf.gz | grep "^#CHROM"

#Filtrar SNPs bialélicos	
bcftools view \
    --type snps \
    --min-alleles 2 \
    --max-alleles 2 \
    --output-type z \
    --output videira_biallelic.vcf.gz \
    videira_joint.vcf.gz

# Indexar
bcftools index --tbi videira_biallelic.vcf.gz

# Confirmar resultado
bcftools stats videira_biallelic.vcf.gz | grep "^SN"

#Conversão para PLINK
plink2 \
    --vcf videira_biallelic.vcf.gz \
    --make-pgen \
    --out videira_raw \
    --threads 4

# Confirmar
wc -l videira_raw.pvar


#Filtros QC
plink2 \
    --pfile videira_raw \
    --maf 0.05 \
    --geno 0.10 \
    --mind 0.20 \
    --hwe 1e-6 \
    --make-pgen \
    --out videira_qc \
    --threads 4

# Ver quantos SNPs sobreviveram
wc -l videira_qc.pvar

# Ver log completo — mostra quantos foram removidos e porquê
cat videira_qc.log

#LD pruning (não é necessário)
plink2 \
    --pfile videira_qc \
    --indep-pairwise 50 5 0.8 \
    --out videira_ld \
    --threads 4

# Ver quantos ficaram
wc -l videira_ld.prune.in

# Aplicar
plink2 \
    --pfile videira_qc \
    --extract videira_ld.prune.in \
    --make-bed \
    --out videira_pruned \
    --threads 4

wc -l videira_pruned.bim

# Converter directamente de pgen para bed (sem LD pruning)
plink2 \
    --pfile videira_qc \
    --make-bed \
    --out videira_pruned \
    --threads 4

# Confirmar
wc -l videira_pruned.bim
cat videira_pruned.log | tail -10

#Kinship com KING
# Instalar KING se necessário
conda install -c bioconda king -y

# Correr kinship
king \
    -b videira_pruned.bed \
    --kinship \
    --prefix videira_king

# Ver resultados
cat videira_king.kin0
